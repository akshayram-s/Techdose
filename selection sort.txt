class Solution {
public:
    vector<int> sortArray(vector<int>& arr) {
         int i,j,min,n=arr.size();
        for(i=0;i<n;i++)
        {
            min = i;
            for(j=i+1;j<n;j++)
                if (arr[j] < arr[min])
                    min = j;
            swap(arr[i],arr[min]);
        }
        return arr;
    }
};
